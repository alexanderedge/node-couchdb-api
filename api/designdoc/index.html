node-couchdb-api The CouchDB API Wrapper for Node.js

Design Document API Official Documentation

var couchdb = require("couchdb-api");

/** default values: "localhost", 5984 **/
var server = couchdb.srv(host, port);

var db = server.db("my-database");

var ddoc = db.ddoc("my-design-doc");
